 > **[[백준] 유기농 배추](https://www.acmicpc.net/problem/1012)**

	* 시간복잡도 : 케이스당 O(M*N)
---------------------

```java
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.StringTokenizer;
import java.io.BufferedWriter;
import java.io.OutputStreamWriter;
import java.util.*;

public class Main{
    static int[] px = {1, -1, 0, 0};
    static int[] py = {0, 0, 1, -1};
    public static void main(String[] args) throws IOException{
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));
        StringBuilder sb = new StringBuilder();
        
        int T = Integer.parseInt(br.readLine());

        StringTokenizer st;
        
        for(int i = 0; i<T; i++){
            st = new StringTokenizer(br.readLine());
            int M = Integer.parseInt(st.nextToken());
            int N = Integer.parseInt(st.nextToken());
            int K = Integer.parseInt(st.nextToken());
            int cnt = 0;
            
            boolean[][] visited = new boolean[N][M];
            
            for(int j = 0; j<K; j++){
                st = new StringTokenizer(br.readLine());
                int X = Integer.parseInt(st.nextToken());
                int Y = Integer.parseInt(st.nextToken());
                visited[Y][X] = true; //배추 표시
            }
            
            
            for(int y = 0; y<N; y++){
                for(int x = 0; x<M; x++){
                    if(visited[y][x]){
                        visited[y][x] = false; //false로 방문 표시
                        bfs(visited, y, x); //bfs가 완료되면 연결영역 모두 방문
                        cnt++; //영역수 카운트
                    }
                }
            }
            sb.append(cnt+"\n");
        }
        
        
        bw.write(sb.toString());
        bw.flush();
        bw.close();
    }
    
    static void bfs(boolean[][] visited, int row, int col){
        Queue<int[]> q = new LinkedList<>();
        q.offer(new int[] {row, col});
        while(!q.isEmpty()){
            int[] out = q.poll();
            
            for(int i = 0; i<4; i++){
                int ny = out[0] + py[i];
                int nx = out[1] + px[i];
                
                if(ny>=0 && ny<visited.length && nx>=0 && nx<visited[0].length){
                    if(visited[ny][nx]){
                        visited[ny][nx] = false;
                        q.offer(new int[] {ny, nx});
                    }
                }
            }
        }
    }
}
```
