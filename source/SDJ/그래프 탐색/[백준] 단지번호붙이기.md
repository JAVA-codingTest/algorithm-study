

> **[[백준] 단지번호붙이기](https://www.acmicpc.net/problem/2667)**

	* 시간복잡도 O(N^2)
	* BFS로 단지의 상하좌우 탐색
---------------------

```java
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.StringTokenizer;
import java.io.BufferedWriter;
import java.io.OutputStreamWriter;
import java.util.*;

public class Main{
    static int N;
    static char[][] map;
    static int[] px = {1, -1, 0, 0}; //x축의 좌우
    static int[] py = {0, 0, 1, -1}; //y축의 상하
    public static void main(String[] args) throws IOException{
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));
        ArrayList<Integer> arr = new ArrayList<>();
        
        N = Integer.parseInt(br.readLine());
        
        map = new char[N][N];
        
        //map 초기화
        for(int i=0; i<N; i++) map[i] = br.readLine().toCharArray();
        
        for(int i=0; i<N; i++){
            for(int j=0; j<N; j++){
                if(map[i][j]!='0'){
                    arr.add(bfs(i, j));
                }
            }
        }
        
        Collections.sort(arr);
        
        StringBuilder sb = new StringBuilder();
        sb.append(arr.size()+"\n");
        
        for(int i : arr) sb.append(i+"\n");
        
        
        bw.write(sb.toString());
        bw.flush();
        bw.close();
    }
    
     /**
	  * bfs탐색을 하기 위해 큐로 실행되는 함수
      * @param y 집의 y축
      * @param x 집의 x축
      */
    static int bfs(int y, int x){
        Queue<int[]> q = new LinkedList<>();
        q.offer(new int[] {y, x});
        map[y][x] = '0';
        int cnt = 1; //첫 방문 집을 q에 삽입했기 때문에 1
        
        while(!q.isEmpty()){
            int[] pop = q.poll();
            
            //py와 px로 상하좌우 탐색
            for(int i=0; i<4; i++){
	            //원래 위치에서 상하좌우 1씩 더한값
                int dy = pop[0] + py[i];
                int dx = pop[1] + px[i];
                
                if(dy<N && dx<N && dy>=0 && dx>=0 && map[dy][dx]=='1'){
                    q.offer(new int[] {dy, dx});
                    map[dy][dx] = '0'; //방문한 집 0으로 초기화
                    cnt++;
                }
            }
        }
        return cnt;
    }
}
```
