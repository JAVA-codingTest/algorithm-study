 > **[[백준] 탈출](https://www.acmicpc.net/problem/3055)**

	* 시간복잡도 : O(R*C)
---------------------

```java
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.StringTokenizer;
import java.io.BufferedWriter;
import java.io.OutputStreamWriter;
import java.util.*;

public class Main{
    static int R, C;
    static char[][] map;
    static boolean[][] visited;
    static int[] px = {1, -1, 0, 0};
    static int[] py = {0, 0, 1, -1};
    static Queue<int[]> water = new LinkedList<>();
    static Queue<int[]> move = new LinkedList<>();
    public static void main(String[] args) throws IOException{
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));
        
        StringTokenizer st = new StringTokenizer(br.readLine());
        
        R = Integer.parseInt(st.nextToken());
        C = Integer.parseInt(st.nextToken());
        
        map = new char[R][C];
        visited = new boolean[R][C];
        
        for(int i = 0; i<R; i++){
            String s = br.readLine();
            for(int j = 0; j<C; j++){
                char c = s.charAt(j);
                map[i][j] = c;
                if(c=='*'){ //처음 물의 위치 
                    water.offer(new int[] {i, j});
                }else if(c=='S'){ //고슴도치 위치
                    move.offer(new int[] {i, j, 0});
                    visited[i][j] = true;
                }
            }
        }

        int res = bfs();
        
        bw.write(res==-1?"KAKTUS":res+"");
        bw.flush();
        bw.close();
    }
    
    //각 날마다 물을 차게 하기 위해 큐 사이즈만큼 bfs 동작
    static void flood(){ 
        int size = water.size();
        for(int i = 0; i<size; i++){
            int[] next = water.poll();
                
            for(int j = 0; j<4; j++){
                int ny = next[0] + py[j];
                int nx = next[1] + px[j];
                
                if(ny>=0 &&nx>=0 && ny<R && nx<C){
                    if(map[ny][nx]=='.'){
                        map[ny][nx] = '*';
                        water.offer(new int[] {ny, nx});
                    }
                }
            }
        }
    }
    
    //각 날마다 이동하기 위해 큐 사이즈만큼 bfs 동작
    static int bfs(){
        while(!move.isEmpty()){
            flood(); //각 날마다 먼저 물이 차오름
            int size = move.size();
            for(int m = 0; m<size; m++){
                int[] next = move.poll();
                    
                for(int i = 0; i<4; i++){
                    int ny = next[0] + py[i];
                    int nx = next[1] + px[i];
                    if(ny>=0 &&nx>=0 && ny<R && nx<C && !visited[ny][nx]){
                        if(map[ny][nx]=='.'){ //방문하지 않았고, 이동할 수 있는 경우
                            visited[ny][nx] = true;
                            move.offer(new int[] {ny, nx, next[2]+1});
                        }else if(map[ny][nx]=='D') return next[2]+1; //다음 날 도착하기 때문에 +1로 반환
                    }
                }
            }
            
        }
        return -1;
    }
}
```
