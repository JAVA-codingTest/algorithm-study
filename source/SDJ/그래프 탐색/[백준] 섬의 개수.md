 > **[[백준] 섬의 개수](https://www.acmicpc.net/problem/4963)**

	* 시간복잡도 : O(w*h)
---------------------

```java
import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.StringTokenizer;
import java.io.BufferedWriter;
import java.io.OutputStreamWriter;
import java.util.*;

public class Main{
    static int[] px = {1, -1, 0, 0, -1, 1, 1, -1};
    static int[] py = {0, 0, 1, -1, -1, 1, -1, 1};
    public static void main(String[] args) throws IOException{
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));
        StringBuilder sb = new StringBuilder();
        
        

        StringTokenizer st;
        
        while(true){
            st = new StringTokenizer(br.readLine());
            int w = Integer.parseInt(st.nextToken());
            int h = Integer.parseInt(st.nextToken());
            if(w==0 && h==0) break;

            int cnt = 0;
            
            int[][] visited = new int[h][w];
            
            for(int i = 0; i<h; i++){
                st = new StringTokenizer(br.readLine());
                for(int j = 0; j<w; j++){
                    visited[i][j] = Integer.parseInt(st.nextToken());
                }
            }
            
            
            for(int y = 0; y<h; y++){
                for(int x = 0; x<w; x++){
                    if(visited[y][x]==1){
                        visited[y][x] = 0; 
                        bfs(visited, y, x);
                        cnt++; 
                    }
                }
            }
            sb.append(cnt+"\n");
        }
        
        
        bw.write(sb.toString());
        bw.flush();
        bw.close();
    }
    
    static void bfs(int[][] visited, int row, int col){
        Queue<int[]> q = new LinkedList<>();
        q.offer(new int[] {row, col});
        while(!q.isEmpty()){
            int[] out = q.poll();
            
            for(int i = 0; i<8; i++){
                int ny = out[0] + py[i];
                int nx = out[1] + px[i];
                
                if(ny>=0 && ny<visited.length && nx>=0 && nx<visited[0].length){
                    if(visited[ny][nx]==1){
                        visited[ny][nx] = 0;
                        q.offer(new int[] {ny, nx});
                    }
                }
            }
        }
    }
}
```
