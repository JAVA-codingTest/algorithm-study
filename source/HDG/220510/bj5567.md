## 결혼식

답안 참고하여 해결했습니다

python3 / 32436kb / 96ms

```python
import collections
import sys

n = int(sys.stdin.readline())
graph = [[] for _ in range(n+1)]
for _ in range(int(sys.stdin.readline())):
    a, b = map(int, sys.stdin.readline().split())
    graph[a].append(b)
    graph[b].append(a)

q = collections.deque()
q.append(1)
dist = [-1] * (n+1)
dist[1] = 0
ret = 0

while q:
    target = q.popleft()
    if 1 <= dist[target] <= 2:
        ret+=1
    if dist[target] == 2:
        continue
    for i in graph[target]:
        if dist[i] != -1:
            continue
        dist[i] = dist[target] + 1
        q.append(i)

print(ret)
```
