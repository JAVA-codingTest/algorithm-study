## 케빈 베이컨의 6단계 법칙

모든 노드에 대해서 bfs하여 해결했습니다

python3 / 32440kb / 88ms

```python
import collections
import sys

n,m = map(int,sys.stdin.readline().split())
graph = [[] for _ in range(n+1)]
for _ in range(m):
    a, b =map(int, sys.stdin.readline().split())
    graph[a].append(b)
    graph[b].append(a)

ans = [0] * (n+1)

for i in range(1,n+1):
    visited = [-1] * (n+1)
    visited[i] = 0
    q = collections.deque()
    q.append(i)

    while q:
        target = q.popleft()
        for j in graph[target]:
            if visited[j] == -1:
                visited[j] = visited[target]+1
                q.append(j)

    ans[i] = sum(visited[1:])

ans = ans[1:]
print(ans.index(min(ans))+1)
```
