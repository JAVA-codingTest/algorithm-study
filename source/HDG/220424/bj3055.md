## 탈출

전에 풀어서 다시 풀었습니다
물과 고슴도치에 대해 큐를 생성하고 
물 기준 bfs 후 고슴도치 기준 bfs 탐색했습니다
고슴도치 기준 bfs 시 물이 먼저 차거나 동시에 접근하는 부분은 제외하고 bfs를 하였습니다.

python3 / 30840kb / 80ms

```python
import sys

r, c = map(int, sys.stdin.readline().split())
hedgehod = []
cave = []
water = []
graph = []
dx = [-1,1,0,0]
dy = [0,0,-1,1]

for i in range(r):
    tmp = list(sys.stdin.readline().split()[0])
    if "D" in tmp:
        cave = [i, tmp.index("D")]
        tmp[tmp.index("D")] = float('inf')
    if "S" in tmp:
        hedgehod.append([i, tmp.index("S")])
        tmp[tmp.index("S")] = 0
    for j in range(len(tmp)):
        if tmp[j] == "*":
            tmp[j] = 0
            water.append([i, j])
        if tmp[j] == "X":
            tmp[j] = -1
    graph.append(list(tmp))

# 물 증가 처리
while water:
    x, y = water.pop(0)
    for i in range(4):
        nx = x + dx[i]
        ny = y + dy[i]
        if 0<=nx<r and 0<=ny<c and graph[nx][ny] == ".":
            graph[nx][ny] = graph[x][y]+1
            water.append([nx,ny])

# 고슴도치 이동 처리
while hedgehod:
    x, y = hedgehod.pop(0)
    for i in range(4):
        nx = x + dx[i]
        ny = y + dy[i]
        if 0 <= nx < r and 0 <= ny < c:
            if graph[nx][ny] == 'D':
                graph[nx][ny] = graph[x][y]
                break
            if graph[nx][ny] == -1:
                continue
            if graph[nx][ny] =="." or (graph[nx][ny] != "X" and graph[nx][ny] > graph[x][y]+1) :
                graph[nx][ny] = graph[x][y] + 1
                hedgehod.append([nx,ny])

print(graph[cave[0]][cave[1]] if graph[cave[0]][cave[1]] != float('inf') else "KAKTUS")
```
