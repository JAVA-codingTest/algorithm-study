import sys

n = int(sys.stdin.readline())
arr = list(map(int, sys.stdin.readline().split()))
arr.sort()

answer = sys.maxsize

for i in range(n-2):
    start = i+1
    end = n-1

    while start < end:
        temp = arr[i] + arr[start] + arr[end]
        if abs(temp) <= answer:
            answer = abs(temp)
            result = [arr[i], arr[start], arr[end]]
        if temp < 0:
            start += 1
        elif temp > 0:
            end-=1
        else:
            break

print(result[0], result[1], result[2])
