```python
import sys

n = int(sys.stdin.readline())
arr = list(map(int, sys.stdin.readline().split()))
arr.sort()
answer = 0

for i in range(n):
    tmp = arr[:i] + arr[i+1:]
    left, right = 0, len(tmp) - 1
    while left < right:
        t = tmp[left] + tmp[right]
        if t == arr[i]:
            answer += 1
            break
        if t < arr[i] :
            left+=1
        else:
            right-=1

print(answer)
```
