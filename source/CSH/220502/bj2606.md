import sys
sys.setrecursionlimit(10**5)
input = sys.stdin.readline

com = int(input())
con = int(input())
vir = [[] for _ in range(com+1)]

for _ in range(con):
    a, b = map(int, input().split())
    vir[a].append(b)
    vir[b].append(a)

def dfs(x):
    visited[x] = True
    for a in vir[x]:
        if visited[a]: continue
        dfs(a)

visited= [False] * (com+1)

dfs(1)

print(visited.count(True) -1) # 1번을 뺀 방문한 곳
