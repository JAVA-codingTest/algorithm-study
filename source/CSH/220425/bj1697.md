# 숨바꼭질

답안 참고하여 수정
```python
import sys
from collections import deque
input = sys.stdin.readline

n, k = list(map(int, input().split()))
X = 100000

dist = [-1] * (X + 1)

def bfs(x):
    queue = deque()
    queue.append(x)
    dist[x] = 0
    def move(y, d):
        if y < 0 or y > X or dist[y] != -1:
            return
        dist[y] = d
        queue.append(y)

    while queue:
        x = queue.popleft()
        move(x*2, dist[x] + 1)
        move(x-1, dist[x] + 1)
        move(x+1, dist[x] + 1)

bfs(n)
print(dist[k])
```

메모리 초과 코드 - 재귀를 너무 많이하는 것이 원인으로 추정

```python
import sys
from collections import deque
input = sys.stdin.readline

X = 100000

n, k = map(int, input().split())

move_lst = []
queue = deque()
visited = [False]*(X+1)

queue.append(n)
visited[n] = True
cnt = 0

def move(x, c, visited):
    
    if x<0 or x>X or visited[x]:
        return
    
    if x == k:
        move_lst.append(c)

    queue.append(x)
        
while queue:
    x = queue.popleft()
    move(x*2, cnt+1, visited)
    move(x-1, cnt+1, visited)
    move(x+1, cnt+1, visited)

print(min(move_lst))
```
